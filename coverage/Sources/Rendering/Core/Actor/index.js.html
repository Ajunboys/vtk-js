export function extend(publicAPI, model, initialValues = {}) {

  // Inheritance

  model.boundsMTime = {};

  // Build VTK API
  macro.set(publicAPI, model, ['property']);

    'backfaceProperty',
    'forceOpaque',
    'forceTranslucent',
    'mapper',
    // 'texture', // Actor should have an array of textures
  ]);
  macro.getArray(publicAPI, model, ['bounds'], 6);

  vtkActor(publicAPI, model);
